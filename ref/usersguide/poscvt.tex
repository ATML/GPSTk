\section{\emph{poscvt}}
\subsection{Usage}
\getuse{poscvt --help}
\subsection{Examples}
\begin{verbatim}
> poscvt --ecef="-4346070.69263 4561978.26297 803.498856837"
\end{verbatim}
\getuse{poscvt --ecef="-4346070.69263 4561978.26297 803.498856837"}
\begin{verbatim}
> poscvt -l
\end{verbatim}
\getuse{poscvt -l}
\begin{verbatim}
> poscvt --ecef="-4346070.69263 4561978.26297 803.498856837" -F "%A %L %h"

  0.007356 133.611574 -77345.241247
\end{verbatim}
